CREATE TALBE icu.inputevents (
    subject_id INT,
    hadm_id INT,
    stay_id INT,
    caregiver_id INT,
    starttime DATETIME,
    endtime DATETIME,
    storetime DATETIME,
    itemid INT,
    amount FLOAT,
    amountuom VARCHAR(10),
    rate FLOAT,
    rateuom VARCHAR(25),
    orderid INT,
    linkorderid INT,
    ordercategoryname VARCHAR(25),
    secondaryordercategoryname VARCHAR(25),
    ordercomponenttypedescription VARCHAR(100),
    ordercategorydescription VARCHAR(25),
    patientweight FLOAT,
    totalamount FLOAT,
    totalamountuom VARCHAR(10),
    isopenbag INT,
    continueinnextdept INT,
    statusdescription VARCHAR(25),
    originalamount FLOAT,
    originalrate FLOAT,
    PRIMARY KEY (subject_id, hadm_id, stay_id, starttime),
    FOREIGN KEY (subject_id) REFERENCES hosp.patients(subject_id) ON DELETE CASCADE,
    FOREIGN KEY (hadm_id) REFERENCES hosp.admissions (hadm_id) ON DELETE CASCADE,
    FOREIGN KEY (stay_id) REFERENCES icu.icustays(stay_id) ON DELETE CASCADE
);
