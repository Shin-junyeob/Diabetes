CREATE TABLE hosp.prescriptions (
    subject_id INT,
    hadm_id INT,
    pharmacy_id INT,
    poe_id VARCHAR(25),
    poe_seq INT,
    order_provider_id VARCHAR(10),
    starttime DATETIME,
    stoptime DATETIME,
    drug_type VARCHAR(10),
    drug VARCHAR(100),
    formulary_drug_cd VARCHAR(25),
    gsn VARCHAR(50),
    ndc VARCHAR(20),
    prod_strength VARCHAR(100),
    form_rx VARCHAR(10),
    dose_val_rx VARCHAR(10),
    dose_unit_rx VARCHAR(10),
    form_val_disp VARCHAR(25),
    form_unit_disp VARCHAR(10),
    doses_per_24_hrs FLOAT,
    route VARCHAR(25),
    PRIMARY KEY (subject_id, hadm_id, pharmacy_id),
    FOREIGN KEY (subject_id) REFERENCES patients(subject_id) ON DELETE CASCADE,
    FOREIGN KEY (hadm_id) REFERENCES admissions (hadm_id) ON DELETE CASCADE
);
